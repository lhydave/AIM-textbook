\begin{document}
\begin{tikzpicture}[thick,>=stealth]
    % 定义坐标轴
    \draw[->] (-3.5, 0) -- (3.5, 0) node[below] {$x$};
    \draw[->] (0, -0.5) -- (0, 4.5) node[right] {$y$};
    
    % 绘制上图区域
      \fill[Orchid, domain=-2:2, samples=100, variable=\x, opacity=0.6] 
      (-2,4) -- plot (\x, {(\x)^2}) -- (2,4) -- cycle;

    % 绘制函数 y = x^2 的图像
    \draw[domain=-2:2, thick, samples=100] plot (\x, {(\x)^2}) node[right] {$f(x) = x^2$};
    
    % 标记 epigraph 区域
    \node at (-1, 3.5) {\large $\epi(f)$};
\end{tikzpicture}
